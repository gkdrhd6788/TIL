```python
#1989
for tc in range(1,int(input())+1):
    str_list = list(input())
    str_length = len(str_list)
    times = str_length//2
    result = 1
    for i in range(times):
        if str_list[i] != str_list[str_length-1-i]:
            result = 0
            break # 생각 못했지만 강의 들으면서 얻은 팁
            # 함수로 짜면 return하면 함수가 바로 끝나기 때문에 중첩 for문이 있는 경우 유용

    print(f'#{tc} {result}')
```
